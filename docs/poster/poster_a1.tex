\begin{column}{0.495\textwidth}
\begin{block}{1. Why Correspondence-Free?}
\justifying
Sweet potato DArT panels use \textbf{disjoint ID namespaces}
(CIP accessions vs.\ plate/well coordinates).
Without an external manifest, cross-panel alignment is infeasible.
We validate each panel \textit{independently} via geometry diagnostics
and robustness curves.

\vspace{1mm}
\centering
{\footnotesize
\begin{tabular}{l r r r}
    \toprule
    \textbf{Panel} & \(\boldsymbol{n}\) & \(\boldsymbol{p}\) & \(\boldsymbol{n/p}\) \\

    \textcolor{okteal}{G-Silico}  & 5\,970 & 57\,715 & 0.10 \\
    \textcolor{okverm}{LD-SNP}    &   630  & 62\,732 & 0.01 \\
    \textcolor{okrose}{LD-Silico} &   635  & 38\,272 & 0.02 \\
    \bottomrule

\end{block}

\begin{column}{0.495\textwidth}
\begin{block}{2. Method (fixed graph protocol)}
\justifying
\begin{itemize}\setlength\itemsep{2mm}
    \item[\textcolor{okblue}{\(\boldsymbol{1}\)}]
          \textbf{PCA-30D} — deterministic analytic space;
          retains \(>\)\,85\% variance across all panels.
    \item[\textcolor{okblue}{\(\boldsymbol{2}\)}]
          \textbf{kNN graph} — \(k\!=\!15\), cosine distance
          on PCA scores; reciprocity \(>\)\,0.64 (all panels).
    \item[\textcolor{okblue}{\(\boldsymbol{3}\)}]
          \textbf{UMAP-2D} — stochastic layout for
          visualisation only (not used in metrics).
    \item[\textcolor{okblue}{\(\boldsymbol{4}\)}]
          \textbf{Validation} — geometry diagnostics
          + robustness perturbations (marker dropout
          5–80\%, MCAR injection +0–20\%).
\end{itemize}
\end{block}

\begin{block}{3. Robustness Under Perturbation}
\begin{columns}[T,totalwidth=\textwidth]

\begin{column}{0.62\textwidth}
\centering
\includegraphics[width=\linewidth]{fig4_robustness_hero.pdf}
\end{column}

\begin{column}{0.36\textwidth}
\justifying

\textbf{Global geometry is invariant.}
Under marker subsampling (5--80\%) and
+0--20\% MCAR injection, the PCA subspace
similarity remains \(\mathrm{SS} \geq 0.91\)
(even at 5\% marker retention) and
\(\mathrm{SS} > 0.998\) under missing injection.
Principal variance structure is genome-wide,
not marker-specific.

\vspace{1mm}

\textbf{Local topology degrades smoothly.}
kNN neighborhood overlap
(\(J_{\text{nbr}}\)) decreases monotonically
from 0.90 to 0.43 under subsampling and
to 0.77--0.86 at +20\% MCAR.

\vspace{1mm}

\textbf{No phase transitions.}
Degradation is continuous across all panels,
enabling curator-defined quality thresholds
without structural collapse.

\vspace{2mm}
\begin{tcolorbox}[colback=white,colframe=blue!40!black,
title=\bfseries Stability Regime]

Let $X \in \mathbb{R}^{n\times p}$ with $n \ll p$.
For random marker subsets $X_s$ with $|s| \geq 0.05p$,

\[
\mathcal{S}_{30}(X) \approx \mathcal{S}_{30}(X_s)
\quad \text{(SS ≥ 0.91)}
\]

Global PCA geometry is stable under extreme
feature dropout, while local neighborhoods
degrade continuously without collapse.

\end{tcolorbox}

\end{column}
\end{columns}
\end{block}

{\fontsize{28}{32}\selectfont\bfseries\color{accent}Synthesis}\par
\vspace{1mm}
\hrule height 0.6pt
\vspace{2mm}

\begin{enumerate}\setlength\itemsep{1mm}
\item \textbf{Subspace invariance:}
PCA-30D geometry remains stable (SS ≥ 0.91) under 95\% marker dropout.

\item \textbf{Topological continuity:}
kNN neighborhoods degrade monotonically without phase transitions.

\item \textbf{Correspondence-free QA:}
Geometry diagnostics + robustness curves provide intra-panel validation without shared identifiers.

\item \textbf{Representation constraint:}
In $n \ll p$ regimes, stability dominates marginal trust gains; PCA remains operationally preferable.
\end{enumerate}

exhibit high inter-seed variance, producing
\textbf{non-reproducible kNN graphs} and unstable
downstream community structure.
