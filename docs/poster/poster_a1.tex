\begin{columns}[T,totalwidth=\textwidth]

\begin{column}{0.495\textwidth}
\begin{block}{1. Why Correspondence-Free?}
\justifying
Sweet potato DArT panels use \textbf{disjoint ID namespaces}
(CIP accessions vs.\ plate/well coordinates).
Without an external manifest, cross-panel alignment is infeasible.
We validate each panel \textit{independently} via geometry diagnostics
and robustness curves.

\vspace{1mm}
\centering
{\footnotesize
\begin{tabular}{l r r r}
    \toprule
    \textbf{Panel} & \(\boldsymbol{n}\) & \(\boldsymbol{p}\) & \(\boldsymbol{n/p}\) \\
    \midrule
    \textcolor{okblue}{G-SNP}     & 5\,970 & 20\,069 & 0.30 \\
    \textcolor{okteal}{G-Silico}  & 5\,970 & 57\,715 & 0.10 \\
    \textcolor{okverm}{LD-SNP}    &   630  & 62\,732 & 0.01 \\
    \textcolor{okrose}{LD-Silico} &   635  & 38\,272 & 0.02 \\
    \bottomrule
\end{tabular}}
\end{block}
\end{column}

\begin{column}{0.495\textwidth}
\begin{block}{2. Method (fixed graph protocol)}
\justifying
\begin{itemize}\setlength\itemsep{1.5mm}
    \item \textbf{PCA-30D} as deterministic analytic space (graph built here)
    \item \textbf{kNN graph}: \(k\!=\!15\), cosine distance on PCA scores
    \item \textbf{UMAP-2D} for visualization only (stochastic)
    \item \textbf{Validation}: geometry diagnostics + robustness perturbations
          (marker dropout 5--80\%, MCAR +0--20\%)
\end{itemize}
\end{block}
\end{column}

\end{columns}

% ROW 2 — ROBUSTNESS HERO  (full width)

\begin{block}{3. Robustness Under Perturbation}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.62\textwidth}
\centering
\includegraphics[width=\linewidth]{fig4_robustness_hero.pdf}
\end{column}
\begin{column}{0.36\textwidth}
\justifying
\textbf{(A) Marker subsampling} (5--80\% retention):
kNN neighborhood overlap (\(J_{\text{nbr}}\)) degrades
smoothly from 0.90 to 0.43 as markers are removed.
The PCA subspace similarity remains \(\geq 0.91\)
even at 5\% retention, confirming that the principal
geometry is genome-wide, not marker-specific.

\vspace{1mm}
\textbf{(B) MCAR injection} (+0--20\%):
\(J_{\text{nbr}}\) drops to 0.77--0.86 at +20\%
additional missing values. The PCA subspace is
practically invariant (SS\,\(>\)\,0.998).

\vspace{1mm}
Degradation is \textbf{smooth and monotonic}
across all four panels --- \textbf{no cliff effects},
enabling data-quality thresholds for germplasm
curation workflows.
\end{column}
\end{columns}
\end{block}

% ═══════════════════════════════════════════════════════════
% ROW 4 — TAKEAWAYS + STABILITY + FAILURE MODE

\begin{columns}[T,totalwidth=\textwidth]

\begin{column}{0.30\textwidth}
\begin{tcolorbox}[takeaway, title={\bfseries Key Takeaways}]
\begin{enumerate}\setlength\itemsep{0.5mm}
    \item \textbf{PCA geometry is robust:} SS\,\(\geq\)\,0.91 under 95\% marker dropout.
    \item \textbf{Monotonic kNN degradation:} no cliff effects; enables
          curator-side quality thresholds.
    \item \textbf{Correspondence-free QA:} geometry diagnostics + robustness curves
          work without shared identifiers.
    \item \textbf{AE not justified in \(n \ll p\):} marginal trust gains vs.\
          2–5\(\times\) worse topological stability.
\end{enumerate}
\end{tcolorbox}
\end{column}

\begin{column}{0.34\textwidth}
\begin{block}{6. Stability Regime}
\centering
\includegraphics[width=\linewidth]{fig_stability_regime.pdf}

\vspace{0.5mm}
\justifying\footnotesize
AE matches PCA trust at \(n/p \gtrsim 0.20\) but the inter-seed
stability gap persists (\(\Delta \approx 0.46\)):
learned embeddings are not operationally justifiable
when \(n/p < 0.3\).
\end{block}
\end{column}

\begin{column}{0.30\textwidth}
\begin{block}{7. Failure Mode}
\justifying
\footnotesize
In ultra-wide regimes (\(n/p < 0.2\)), learned embeddings
exhibit high inter-seed variance, producing
\textbf{non-reproducible kNN graphs} and unstable
downstream community structure.

\vspace{1mm}
\centering
{\small
\begin{tabular}{@{} l c @{}}
    \toprule
    \textbf{Indicator} & \(n/p < 0.2\) \\
    \midrule
    AE seed variance    & \textcolor{okverm}{high} \\
    Graph topology      & \textcolor{okverm}{unstable} \\
    Clustering reprod.  & \textcolor{okverm}{non-repr.} \\
    \bottomrule
\end{tabular}}

\vspace{1mm}
\justifying
Stability, not trust, is the \textbf{binding operational
constraint}.
\end{block}
\end{column}
